import java.util.*;
class Main{
	public static void main(String args[]){
		Scanner scan = new Scanner(System.in);
		int choice = 1; 
		
		do{ // for entering multiple students
			System.out.println("Enter Name:");
			String name = scan.nextLine();
		
			System.out.println("Enter PRN:");
			int prn = Integer.parseInt(scan.nextLine());
		
			System.out.println("Enter Batch:");
			String batch = scan.nextLine();
		
			System.out.println("Enter CGPA:");
			float cgpa = Float.parseFloat(scan.nextLine());
		
			StudentOperations operations = new StudentOperations();
			Student student = new Student(name,prn,batch,cgpa);
		
			operations.addStudent(student);
			operations.displayStudents();
		
			System.out.println("Enter choice: 1.Continue 2.Exit");
			choice = Integer.parseInt(scan.nextLine()); // converts string input into float
			// we dont use scanFloat since it can cause issues with the input values after the current input.
		}while(choice != 2);
		
		
	}
}
